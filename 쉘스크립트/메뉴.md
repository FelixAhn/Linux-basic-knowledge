## 메뉴 구성 및 스크립트 구성
1. uptime
2. df -h
3. free -m
4. pstree
5. 서버를 입력받아 ssh로 접속
6. 6.메뉴 종료

* 함수 menu
  * clear
  * txt로 메뉴를 출력
* 함수 press_key
  * 메뉴 선택해서 실행 이후에 일시중지 (Press any key to continue...)

## 메뉴 실습
```bash
#!/bin/bash

function menu {
    clear
    cat << EOF
    ========== 메뉴 ==========
    1. 로드 에버리지
    2. 디스크 상태
    3. 메모리 상태
    4. 프로세스 트리
    5. 입력한 서버에 ssh접속
    6. 메뉴 종료
EOF

read -p "원하는 메뉴 번호를 선택 : " SELECT
}

function press_key{
    echo
    read -n1 -rsp "Press an key to continue..."
    echo
    echo
}

while true
do
    menu
    case ${SELECT} in
        1) 
            clear
            echo "로드 에버리지"
            uptime
            press_key
            ;;
        2) 
            clear
            echo "디스크 상태"
            df -h
            press_key
            ;;
        3) 
            clear
            echo "메모리 상태"
            free -m
            press_key
            ;;
        4) 
            clear
            echo "프로세스 트리"
            pstree
            press_key
            ;;
        5) 
            clear
            read -p "접속 하고 싶은 서버 이름 입력: " SVR
            sleep 1
            echo "${SVR}에 접속중..."
            ssh SVR
            ;;
        6) 
            exit 0
            ;;
        *)
            echo "번호를 잘못 입력했습니다."
            press_key
            ;;
    esac
done
```

## 메뉴 실습 (GUI 버전)
```bash

    dialog --title "GUI MENU" \
            --radiolist "메뉴를 선택해주세요" 20 35 6 \

            2 "디스크 상태" off \
            3 "메모리 상태" off \
            4 "프로세스 트리" off \
            5 "서버 접속" off \
            6 "메뉴 종료" off \
            2>./select
}

    echo
    read -n1 -rsp "Press an key to continue..."
    echo
    echo
}

while true
do
    menu
    SELECT=$(cat select)
    case ${SELECT} in
        1) 
            clear
            echo "로드 에버리지"
            uptime
            press_key
            ;;

            clear
```
